package com.day1.lab1;

public class PalindromeORnot {

    public static boolean checkWord(String word) {
        String wordToCheck = word.toLowerCase(); 

        int length = wordToCheck.length();

        for (int i = 0; i < length / 2; i++) {
            if (wordToCheck.charAt(i) != wordToCheck.charAt(length - 1 - i)) {
                return false;
            }
        }
        return true;
    }

    public static void main(String[] args) {
        String testInput1 = "level";
        String testInput2 = "happy";

        System.out.println("Is '" + testInput1 + "' a palindrome? " + checkWord(testInput1));
        System.out.println("Is '" + testInput2 + "' a palindrome? " + checkWord(testInput2));
    }
}
